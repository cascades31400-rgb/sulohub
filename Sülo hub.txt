UserInputService.JumpRequest:Connect(function()
    if _G.InfJump and LocalPlayer.Character then LocalPlayer.Character.Humanoid:ChangeState("Jumping") end
end)

NoclipBtn.MouseButton1Click:Connect(function()
    _G.Noclip = not _G.Noclip
    NoclipBtn.Text = "Noclip: " .. (_G.Noclip and "ON" or "OFF")
end)

RunService.Stepped:Connect(function()
    if _G.Noclip and LocalPlayer.Character then
        for _, v in pairs(LocalPlayer.Character:GetDescendants()) do
            if v:IsA("BasePart") then v.CanCollide = false end
        end
    end
end)

SpeedBtn.MouseButton1Click:Connect(function()
    if LocalPlayer.Character and LocalPlayer.Character.Humanoid then
        LocalPlayer.Character.Humanoid.WalkSpeed = (LocalPlayer.Character.Humanoid.WalkSpeed == 16 and 100 or 16)
    end
end)

ToggleBtn.MouseButton1Click:Connect(function()
    _G.ESP_Enabled = not _G.ESP_Enabled
    ToggleBtn.BackgroundColor3 = _G.ESP_Enabled and Color3.fromRGB(0, 150, 0) or Color3.fromRGB(180, 40, 40)
end)

local function CreateESP(P)
    local B = Drawing.new("Square"); B.Color = Color3.fromRGB(255, 200, 0); B.Thickness = 1; table.insert(ESP_Objects, B)
    RunService.RenderStepped:Connect(function()
        if _G.SuloActive and _G.ESP_Enabled and P.Character and P.Character:FindFirstChild("HumanoidRootPart") and P ~= LocalPlayer then
            local Pos, OnS = Camera:WorldToViewportPoint(P.Character.HumanoidRootPart.Position)
            if OnS then B.Size = Vector2.new(2000/Pos.Z, 3000/Pos.Z); B.Position = Vector2.new(Pos.X - B.Size.X/2, Pos.Y - B.Size.Y/2); B.Visible = true else B.Visible = false end
        else B.Visible = false end
    end)
end

TPBtn.MouseButton1Click:Connect(function()
    local target = PlayerInput.Text:lower()
    for _, v in pairs(Players:GetPlayers()) do
        if v.Name:lower():sub(1,#target) == target then LocalPlayer.Character.HumanoidRootPart.CFrame = v.Character.HumanoidRootPart.CFrame end
    end
end)

for _, v in pairs(Players:GetPlayers()) do CreateESP(v) end
Players.PlayerAdded:Connect(CreateESP)
